factory RLPList.fromBuffer(
  Uint8List buffer,
  int offset,
  int childOffset,
  int length,
) {
  final items = <RLPItem>[];
  while (childOffset < offset + 1 + length) {
    final (item, consumed) = decodeRLP(buffer, offset: childOffset);
    items.add(item);
    childOffset += consumed;
  }
  return RLPList(items);
}
